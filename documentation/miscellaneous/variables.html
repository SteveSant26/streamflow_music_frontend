src/app/shared/utils/auth-interceptor.ts

authInterceptor
Type : HttpInterceptorFn
Default value : (req, next) => { const authService = inject(AuthService); const token = authService.session()?.access_token; if (token) { const cloned = req.clone({ headers: req.headers.set("Authorization", `Bearer ${token}`), }); return next(cloned); } return next(req); }
