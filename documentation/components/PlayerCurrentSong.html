import { ChangeDetectionStrategy, Component, Input } from "@angular/core";
import { MatIconModule } from "@angular/material/icon";
import { Router } from "@angular/router";

interface Song {
  id: number;
  title: string;
  artists: string[];
  album: string;
  albumId: number;
  duration: string;
  image: string;
}

@Component({
  selector: "app-player-current-song",
  imports: [MatIconModule],
  templateUrl: "./player-current-song.html",
  changeDetection: ChangeDetectionStrategy.OnPush,
})
export class PlayerCurrentSong {
  @Input() song: Song | null = null;

  constructor(private readonly router: Router) {}

  navigateToCurrentSong() {
    if (this.song) {
      this.router.navigate(["/currentSong"]);
    }
  }
}

@if (song) {
<button
  class="flex items-center gap-2 sm:gap-3 cursor-pointer hover:bg-gray-100 rounded-lg p-1 sm:p-2 transition-colors group w-full text-left border-none bg-transparent"
  (click)="navigateToCurrentSong()"
  type="button"
  aria-label="Ver canción actual en pantalla completa"
>
  <picture class="w-10 h-10 sm:w-14 sm:h-14 flex-shrink-0">

}
